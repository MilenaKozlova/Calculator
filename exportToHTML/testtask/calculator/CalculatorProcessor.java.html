16   public class CalculatorProcessor {
17         static int process(String inputString) throws Exception {
18            ParcerStruct struct = new ParcerStruct();
19            Integer firstNumber;
20            Operations operation;
21            Integer secondNumber;
22            
23            switch(ParcerHelper.GetNumbersType(inputString)){
24                case Arabic -> {                
25                   firstNumber = ParcerHelper.parceArabicNumber(inputString, struct);        
26                   operation = ParcerHelper.parceOperation(inputString, struct);
27                   secondNumber = ParcerHelper.parceArabicNumber(inputString, struct);             
28                }
29                case Roman -> {
30                   firstNumber = ParcerHelper.parceRomanNumber(inputString, struct);        
31                   operation = ParcerHelper.parceOperation(inputString, struct);
32                   secondNumber = ParcerHelper.parceRomanNumber(inputString, struct);              
33                }
34                default -> throw new Exception("Entered data is incorrect");        
35            } 
36            
37           validateInputData(struct, inputString, firstNumber, secondNumber);
38           return process(firstNumber, secondNumber, operation);
39       }    
40       static int process(Integer firstNumber, Integer secondNumber, Operations operation){
41           Integer result = 0;
42           switch(operation){
43               case Sum:
44                   result = firstNumber + secondNumber;
45                   break;
46               case Sub:
47                   result = firstNumber - secondNumber;
48                   break;
49               case Mul:
50                   result = firstNumber * secondNumber;
51                   break;
52               case Div:
53                   result = firstNumber / secondNumber;
54                   break;        
55           }
56           return result;
57       }  
58       static void validateInputData(ParcerStruct struct, String inputString,Integer firstNumber,Integer secondNumber) throws Exception {
59           validateNumber(firstNumber);
60           validateNumber(secondNumber);
61           if(struct.startPosition >  inputString.length()){
62               throw new Exception("Entered data is incorrect");
63           }    
64        
65       }
66       static void validateNumber(Integer number) throws Exception {
67          if(number == null || number > 10 || number < 1){
68               throw new Exception("Entered data is incorrect");
69          }
70       }
71               
72               
73   }
